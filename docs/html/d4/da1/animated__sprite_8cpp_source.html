58 void AnimatedSprite::initAnimations( const TMXLoader& loader )
59 {
60 
61  // Recebemos os vetores que foram carregados pelo loader
62  const vector< TMXTileSet* >& tmx_tilesets = loader.getTmxTilesets();
63  const vector< TMXLayer* >& tmx_layers = loader.getTmxLayers();
64 
65  // Criamos um vetor para receber as ImageResource que servirao
66  // como base para os tiles
67  ImageResource* baseImage[ tmx_tilesets.size() ];
68 
69  // Carregamos as imagens
70  for( unsigned int i = 0; i < tmx_tilesets.size(); i++ ) {
71 
72  try {
73  // Criamos a imageResource
74  baseImage[i] = ImageResource::loadImageResource( tmx_tilesets[i]->getSource() );
75 
76  } catch( Ludic::Exception& ex ) {
77  cout << ex.what() << endl;
78  return;
79  }
80 
81  // Setamos a colorkey da imagem, se houver
82  if( !tmx_tilesets[i]->getColorkey().empty() ) {
83  baseImage[i]->setColorKey( Color( tmx_tilesets[i]->getColorkey() ) );
84  }
85 
86  }//for i
87 
88  // Criamos o animation
89  Animation* anim = nullptr;
90 
91  for( unsigned int i = 0; i < tmx_layers.size(); i++ ) {
92 
93  // Recebemos o vetor de dados do tmx_layer
94  const vector< TMXLayer::DataInfo >& data = tmx_layers[i]->getData();
95 
96  // Carregamos a animation
97  anim = new Animation( tmx_tilesets, data, baseImage, tmx_layers[i]->getName() );
98 
99  // Armazenamos o novo animation
100  animationMap[ tmx_layers[i]->getName() ] = anim;
101 
102  }//for i
103 
104  // Setamos a animation adicionada com animation atual
105  currentAnimation = anim;
106 
107  // Ajustamos as dimensões do retangulo de colisão
108  collisionBox.setDimension( anim->getFrameWidth(), anim->getFrameHeight() );
109
